Echart散点图自定义图标

Feb 7, 2021 16:06 · 166 words · 1 minute read ECharts

参数如下：

option = {
    title: {
        text: '对数轴示例',
        left: 'center'
    },
    tooltip: {
        trigger: 'axis',
        axisPointer: {
            type: 'cross'
        },
        formatter: '{a0}: {c0}<br />{a1}: {c1}<br/>'
    },
    legend: {
        left: 'left',
        data: ['2的指数', '3的指数']
    },
    xAxis: {
        type: 'category',
        name: 'x',
        splitLine: {show: false},
        data: ['一', '二', '三', '四', '五', '六', '七', '八', '九']
    },
    grid: {
        left: '3%',
        right: '4%',
        bottom: '3%',
        containLabel: true
    },
    yAxis: {
        type: 'log',
        name: 'y',
        minorTick: {
            show: true
        },
        minorSplitLine: {
            show: true
        }
    },
    series: [
        {
            name: '3的指数',
            type: 'line',
            data: [1, 3, 9, 27, 81, 247, 741, 2223, 6669]
        },
        {
            tooltip:{
                position: [10,10],
                formatter: 'dfjkdjf'
            },
            name: '2的指数',
            type: 'line',
            data: [1, 2, 4, 8, 16, 32, 64, 128, 256]
        },
        {
            tooltip:{
                trigger: 'item',
                position: function (point, params, dom, rect, size) {
                      // 固定在顶部
                      return [point[0], '90%'];
                 },
                formatter: function(params) {
                    return params.name + params.value
                }
            },
            symbol: "image://https://company-huanyu.oss-cn-beijing.aliyuncs.com/download.jpg",
            symbolSize: 20,
            name: '1/2的指数',
            type: 'scatter',
            data: [{value:1, name:"涨价"}, {}, {value:1, name:"fdjk"}]
        }
    ]
};
